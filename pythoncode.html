Programa para Descargar Vídeos de YouTube

	  	#pip install pytube
from pytube import YouTube
link = input("Introduce el link del vídeo: ")
yt = YouTube(link)
ys = yt.streams.get_highest_resolution()
ys.download()

Programa para Pasar de Texto a Voz y Descargarlo

	  	#pip install gtts
from gtts import gTTS
import os
texto = str(input("Introduce un texto: "))
tts = gTTS(texto, lang='es-es')
tts.save("good.mp3")
